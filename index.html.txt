    <script>
        let balance = parseFloat(localStorage.getItem('earningsBalance')) || 0;
        let transactions = JSON.parse(localStorage.getItem('earningsTransactions')) || [];

        function updateUI() {
            document.getElementById('balance').textContent = `$${balance.toFixed(2)}`;
            document.getElementById('balance').classList.add('pulse-animate');
            setTimeout(() => {
                document.getElementById('balance').classList.remove('pulse-animate');
            }, 600);
            
            const progress = (balance / 25) * 100;
            document.getElementById('progress').style.width = `${Math.min(progress, 100)}%`;
            
            const withdrawBtn = document.getElementById('withdrawBtn');
            withdrawBtn.disabled = balance < 25;
            
            renderHistory();
        }

        function addEarning(amount) {
            balance += amount;
            localStorage.setItem('earningsBalance', balance);
            
            transactions.push({
                type: getTransactionType(amount),
                amount: amount,
                time: new Date().toLocaleString()
            });
            localStorage.setItem('earningsTransactions', JSON.stringify(transactions));
            
            updateUI();
        }

        function getTransactionType(amount) {
            if (amount === 1.00) return 'Like';
            if (amount === 0.50) return 'Subscribe';
            if (amount === 0.25) return 'Watch';
            return 'Other';
        }

        function renderHistory() {
            const container = document.getElementById('history');
            if (transactions.length === 0) {
                container.innerHTML = '<div class="text-gray-500 text-center py-4">No transactions yet</div>';
                return;
            }
            
            container.innerHTML = '';
            transactions.slice().reverse().forEach(tx => {
                const div = document.createElement('div');
                div.className = 'flex justify-between items-center py-2 border-b border-gray-100';
                div.innerHTML = `
                    <div>
                        <span class="font-medium text-gray-700">${tx.type}</span>
                        <span class="text-gray-500 text-sm ml-2">${tx.time}</span>
                    </div>
                    <span class="font-semibold text-green-600">+\$${tx.amount.toFixed(2)}</span>
                `;
                container.appendChild(div);
            });
        }

        document.getElementById('withdrawBtn').addEventListener('click', function() {
            if (balance >= 25) {
                const amount = balance;
                balance = 0;
                localStorage.setItem('earningsBalance', 0);
                localStorage.setItem('earningsTransactions', JSON.stringify([]));
                transactions = [];
                updateUI();
                alert(`Successfully withdrawn $${amount.toFixed(2)}!`);
            }
        });

        // Initial load
        updateUI();
    </script>
